def handle(path:str):
    print("<<<<<<<<",path,">>>>>>>")
def run_all(path:str):
    tr = walk(path)
    for t in tr:
        for file_ in t[2]:
            handle(file_)
def run_all_opt(path:str):
    ZZ = [ t[2] for t in walk(path)]
    [handle(file_) for file_ in ZZ]
